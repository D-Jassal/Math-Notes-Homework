\subsection{Similarity}
\textit{Recall. } Let $A,B\in M_n(F)$.
\defn{Similar Matrices}{
    Let $A,B\in M_n(F)$. Then, $A,B$ are called similar is there exists an invertible matrix $P$ such that $B=P^{-1}AP$ ($A=PBP^{-1}$). 
}
\ex{
    If $A,B$ are similar then, 
    \begin{enumerate}
        \item $\tr(A)=\tr(B)$
        \item $\chi_A(x)=\chi_B(x)$
        \item $\rank(A)=\rank(B)$
    \end{enumerate}
}
\ex{
    \[
        V_\beta\overset{T}{\to} V_\beta
    \]
    vs
    \[
        V_\gamma\overset{T}{\to} V_\gamma.
    \]
    How are $\tensor[_{\beta}]{[T]}{_{\beta}}$ and $\tensor[_{\gamma}]{[T]}{_{\gamma}}$ related?
    \begin{align*}
        \tensor[_{\gamma}]{[T]}{_{\gamma}}&=\tensor[_{\gamma}]{[I_V]}{_{\beta}}\tensor[_{\beta}]{[T]}{_{\beta}}\tensor[_{\beta}]{[I_V]}{_{\gamma}}\\
        &=\tensor[_{\beta}]{[I_V]}{_{\gamma}}^{-1}\tensor[_{\beta}]{[T]}{_{\beta}}\tensor[_{\beta}]{[I_V]}{_{\gamma}}\\
        &=P^{-1}AP.
    \end{align*}
}
\begin{claim}
Also, if $B=P^{-1}AP$ then there are basis' of $V$ such that $A=\tensor[_{\beta}]{[T]}{_{\beta}}$ and $B=\tensor[_{\gamma}]{[T]}{_{\gamma}}$
\end{claim}
\pf[Proof of Claim]{
    Let $T=T_A$ , $TX=AX$, $T\in\mathcal{L}(F^n,F^m)$, $A_{n\times m}X\in F^n$.
    \[
        F^n_\beta \underset{\tensor[_{\beta}]{[I_{T_V}]}{_{e}}}{\overset{I_V}{\to}} F^n_e \underset{A}{\overset{TX}{\to}} F^n_e \underset{\tensor[_{e}]{[I_{F^n}]}{_{\beta}}}{\overset{I_{F^n}}{\to}} F_\beta    
    \]
    Chose $\beta=p_1,\dots,p_n$ so $P=[p_1,\dots,p_n]$. Then $\tensor[_e]{[I_{F^n}]}{_\beta}=P$, and $P=\tensor[_e]{[I_{F^n}]}{_\beta}$.
}
\pf[Proof of Properties of Similar Matrices]{
    \begin{enumerate}
        \item $\tr(A)=\tr(B)$.\\
        The proof is based on $\tr(CD)=\tr(DC)$
        \begin{align*}
            \tr(P^{-1}AP)&=\tr(P^{-1}AP)\\
            &=\tr(APP^{-1})\\
            &=\tr(A)
        \end{align*}
        \item $\chi_A(x)=\chi_B(x)$
        \[
            \chi_A(x)=\det(xI-A)
        \]
        \[
            \chi_B(x)=\det(xI_P^{-1}AP)
        \]
        but $\det(P(xI_P^{-1}AP)P^{-1})=\det(xI_P^{-1}AP)\det(P^{-1})\det(P)=\chi_B(x)$, which is also $\det((PxI_P^{-1}AP)P^{-1})=\det((xIP-AP)P^{-1})=\det(xI-A)=\chi_A(x).$
    \end{enumerate}
}
