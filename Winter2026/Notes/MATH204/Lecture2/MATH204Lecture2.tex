\textit{Ex 22.}
\[
    \begin{cases}
        x(t)=\frac{1}{\sqrt{t+1}},\\
        y(t)=\frac{t}{t+1};\quad t>-1.
    \end{cases}
\]
Convert into rectangular form.
\[
    x^2=\frac{1}{t+1}\Rightarrow t+1=\frac{1}{x^2}\Rightarrow t=\frac{1}{x^2}-1=\frac{1-x^2}{x^2}
\]
\[
    y=\frac{\frac{1-x^2}{x^2}}{\frac{1}{x^2}}=1-x^2;\quad\frac{1}{x^2}\geq 0
\]

\subsection*{Hyperbolic Functions}
The hyperbolic functions are $\sinh$, $\cosh$ and $\tanh$. We define them as
\[
    \sinh x=\frac{e^x-e^{-x}}{2},
\]
\[
    \cosh x=\frac{e^x+e^{-x}}{2},
\]
and
\[
    \tanh x =\frac{\sinh x}{\cosh x}=\frac{e^x-e^{-x}}{e^x+e^{-x}}.
\]
Some indentities involving the hyperbolic functions
\[
    \cosh^2x+\sinh^2x=1
\]
\[
    \cosh x-\sinh x =\frac{e^x+e^{-x}-e^x+e^{-x}}{2}=e^{-x}
\]
which implies that
\[
\lim_{x\to\infty}(\cosh x-\sinh x)=0.
\]
This means that the two functions are asymptotic for $x>>0$.\\
\textit{Note.}
\[
    \sinh(-x)=-\sinh x
\]
is an odd function.
\[
    \cosh(-x)=\cosh x
\]
is an even function.
\[
    \tanh(-x)=-\tanh x
\]
is an odd function.
\[
    \lim_{x\to-\infty}\sinh x=-\infty;\quad \lim_{x\to\infty}\sinh x=\infty;
\]
\[
    \lim_{x\to-\infty}\cosh x=\infty;\quad \lim_{x\to\infty}\cosh x=\infty;
\]
\[
    \lim_{x\to-\infty}\tanh x=\lim_{x\to-\infty}\frac{e^{-x}(e^{2x}-1)}{e^{-x}(e^{2x}+1)}=-1;\quad \lim_{x\to\infty}\tanh x=\lim_{x\to\infty}\frac{e^{x}(1-e^{-2x})}{e^{x}(1+e^{-2x})}=1.
\]
\subsection*{Hyperbolic Derivatives}
\[
\frac{d}{dx}\sinh x=\frac{e^x+e^{-x}}{2}=\cosh x,
\]
\[
    \frac{d}{dx}\cosh x=\frac{e^x-e^{-x}}{2}=\sinh x,
\]
\[
    \frac{d}{dx}\tanh x=\frac{\cosh^2x-\sinh^2x}{\cosh^2x}=\frac{1}{\cosh^2x}.
\]
tanh$x$ admits $y=1$ as a horizontal asymptote at $x=\infty$ and $y=-1$ as a horizontal asymptote at $x=-\infty$. tanh$x$ is an increasing function.
