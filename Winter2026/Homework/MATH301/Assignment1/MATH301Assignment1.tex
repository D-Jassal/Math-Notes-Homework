\section*{Problem 1}
Express in the form $a+bi$, $a,b\in\R$
\begin{enumerate}[label=(\alph*)]
    \item $(2+i)^4$
    \item $\frac{3+4i}{4+3i}$
\end{enumerate}
\stepcounter{section}
\section*{Problem 2}
Find the three roots of $z^3-4=0$.
\stepcounter{section}
\section*{Problem 3}
Assuming that either $|z| = 1$ or $|w| = 1$ and $\overline{z}w\neq1$, prove that
\[
    \left|\frac{z-w}{1-\overline{z}w}\right|=1.
\]
\stepcounter{section}
\section*{Problem 4}
Use complex numbers to prove identity:
\[
    1+\cos(\theta)+\cos(2\theta)+\cdots+\cos(n\theta)=\frac{1}{2}+\frac{\sin\left(n+\frac{1}{2}\theta\right)}{2\sin\left(\frac{\theta}{2}\right)}.
\]
\stepcounter{section}
\section*{Problem 5}

\section*{Problem 6}

\stepcounter{section}
\section*{Problem 7}

\stepcounter{section}
\section*{Problem 8}

\stepcounter{section}
\section*{Problem 9}
